CREATE DATABASE IF NOT EXISTS lopushok_db CHARACTER SET utf8mb4 COLLATE utf8mb4_unicode_ci;
USE lopushok_db;

CREATE TABLE Role (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Title VARCHAR(100) NOT NULL
);

CREATE TABLE User (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Login VARCHAR(100) NOT NULL,
    Password VARCHAR(100) NOT NULL,
    RoleID INT NOT NULL,
    FOREIGN KEY (RoleID) REFERENCES Role(ID)
);

CREATE TABLE ProductType (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Title VARCHAR(100) NOT NULL
);

CREATE TABLE Product (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Title VARCHAR(100) NOT NULL,
    ProductTypeID INT NOT NULL,
    ArticleNumber VARCHAR(50) NOT NULL,
    Description TEXT,
    Image VARCHAR(255),
    ProductionPersonCount INT,
    ProductionWorkshopNumber INT,
    MinCostForAgent DECIMAL(10, 2) NOT NULL,
    FOREIGN KEY (ProductTypeID) REFERENCES ProductType(ID)
);

CREATE TABLE Material (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    Title VARCHAR(100) NOT NULL
);

CREATE TABLE ProductMaterial (
    ProductID INT NOT NULL,
    MaterialID INT NOT NULL,
    Count FLOAT NOT NULL,
    PRIMARY KEY(ProductID, MaterialID),
    FOREIGN KEY (ProductID) REFERENCES Product(ID) ON DELETE CASCADE,
    FOREIGN KEY (MaterialID) REFERENCES Material(ID) ON DELETE CASCADE
);
